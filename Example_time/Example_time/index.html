    rho = np.sqrt(0.7)  # AR(1) coefficient #0.7

    def RMSE(x,y):
    x = x.reshape(-1)
    y = y.reshape(-1)
    n = x.shape[0]
    return(np.sqrt(np.sum(np.square(x-y))/n))
